	int N, A=0, D=0;
    cin >> N;
    char arr[N];
    for (int i = 0; i < N; i++) cin >> arr[i];
    for (int i = 0; i < N; i++) {
    	if (arr[i] == 'A') A++;
    	else if (arr[i] == 'D') D++;
    }

    if (A == D) cout << "Friendship";
    else if (A  > D) cout << "Anton";
    else cout << "Danik";

    return 0;
	
}
