#include <bits/stdc++.h>
using namespace std;

int main() {

	#ifndef ONLINE_JUDGE
		freopen("input.txt", "r", stdin);
		freopen("output.txt", "w", stdout);
	#endif

	std::ios::sync_with_stdio(0);
	cin.tie(0);

	int k, n, w, amount=0;
	cin >> k >> n >> w;

	for (int i = 1; i <= w; i++) {
		amount += i*k;
	}

	if (n > amount) cout << 0 << "\n";
	else cout << amount-n << "\n";
    
    return 0;
	
}
